colramp<-colorRampPalette(rev(brewer.pal(n = 7, name ="RdBu")))(100)

#sort Cols
meta.all<-colData(rld) %>% 
    as.tibble() 

# split groups
id.list<-split(meta.all$id, 
      as.factor(paste(meta.all$genotype,meta.all$treatment, sep = ":")))
#get median by group
mat.median<-sapply(id.list, function(x) apply(mat[,x],1,median))
#z-scores
mat.median.z<-t(apply(mat.median,1,scale))
colnames(mat.median.z)<-colnames(mat.median)

#sort by genotype
id.Ctrl<-meta.all %>% filter(genotype=="Ctrl") %>% pull(id)
id.C118F<-meta.all %>% filter(genotype=="C118F") %>% pull(id)
id.H77L<-meta.all %>% filter(genotype=="H77L") %>% pull(id)

#sort by treatment
id.Ctrl.s<-id.Ctrl[meta.all %>% filter(id %in% id.Ctrl) %>% pull(treatment) %>% order()]
id.C118F.s<-id.C118F[meta.all %>% filter(id %in% id.C118F) %>% pull(treatment) %>% order()]
id.H77L.s<-id.H77L[meta.all %>% filter(id %in% id.H77L) %>% pull(treatment) %>% order()]

ordCol<-c(id.Ctrl.s,id.H77L.s, id.C118F.s)

mat.all<-mat[,ordCol]
